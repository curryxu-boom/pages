# linux中mysql安装

# 准备工作

  • 操作系统：Linux CentOS 7.x

  • MySQL安装包：在官网下载需要的版本，地址：https://dev.mysql.com/downloads/mysql/

  • 注：1.该链接点进去默认是最新版本，可点击Archives按钮选择历史版本

    ​ 2.点击"download"会让你登录/注册或者点击下方小字 -> "No thanks,just start my download"

# 1、将gz文件移动到/usr/local目录

mv mysql-5.6.33-linux-glibc2.5-x86_64.tar.gz /usr/local/
1

# 2、创建mysql目录并将gz解压

mkdir mysql
cd /usr/local
mv mysql-5.6.33-linux-glibc2.5-x86_64.tar.gz mysql
cd mysql
tar -zxvf mysql-5.6.33-linux-glibc2.5-x86_64.tar.gz
1

# 3、创建用户和组

#添加用户组
groupadd mysql
#添加用户mysql 到用户组mysql
useradd -r -g mysql mysql 或 
useradd -r -g mysql -s /bin/false mysql

# 4、更改根目录的拥有者

cd mysql/

# 5、5.7.39步骤 修改配置文件my.cnf

vim /etc/my.cnf
1
[mysqld]
server_id=1
log-bin=mysql-bin
binlog-format=ROW
basedir=/usr/local/mysql/mysql-5.7.39-linux-glibc2.12-x86_64
datadir=/usr/local/mysql/mysql-5.7.39-linux-glibc2.12-x86_64/data
port=3306
sql_mode=NO_ENGINE_SUBSTITUTION,STRICT_TRANS_TABLES
symbolic-links=0
max_connections=600
innodb_file_per_table=1
#忽略大小写
lower_case_table_names=1
character_set_server=utf8
pid-file=/usr/local/mysql/mysql-5.7.39-linux-glibc2.12-x86_64/data/mysql.pid
#socket=/var/lib/mysql/mysql.sock
# Disabling symbolic-links is recommended to prevent assorted security risks
symbolic-links=0
# Settings user and group are ignored when systemd is used.
# If you need to run mysqld under a different user or group,
# customize your systemd unit file for mariadb according to the
# instructions in http://fedoraproject.org/wiki/Systemd

[mysqld_safe]
#log-error=/var/log/mariadb/mariadb.log
#pid-file=/usr/local/mysql/mysql-5.7.39-linux-glibc2.12-x86_64/data/mysql.pid

#
# include all files from the config directory
#
!includedir /etc/my.cnf.d

# 6、5.7.39步骤 在/usr/local/mysql 目录下创建一个 data 文件夹

cd /usr/local/mysql
mkdir data
1

# 7、5.7.39步骤 进入mysql的bin目录初始化

cd /usr/local/mysql/bin/
# 初始化
./mysqld --initialize --user=mysql --defaults-file=/etc/my.cnf --basedir=/usr/local/mysql/ --datadir=/usr/local/mysql/data/

如果出现以下错误
　　./bin/mysqld: error while loading shared libraries: libaio.so.1: cannot open shared object file: No such file or directory
　　执行： yum install -y libaio

# 9、5.7.39步骤 将mysql加入到可控制启动服务的文件夹内，并命名mysql

（目的：可用 service mysql start命令来控制启动 mysql）

先将mysql.server放置到/etc/init.d/mysql中
cd /usr/local/mysql/support-files

cp /usr/local/mysql/support-files/mysql.server /etc/init.d/mysql

cp mysql.server /etc/init.d/mysql

# 11、进入 /etc/my.cnf文件里面添加如下内容

# 14、登陆MySQL

# 15、5.7.39步骤 修改密码

依次执行下面三步
SET PASSWORD = PASSWORD('123456');
ALTER USER 'root'@'localhost' PASSWORD EXPIRE NEVER;
FLUSH PRIVILEGES;
