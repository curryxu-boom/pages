select empno,ename,job,(sal+nvl(comm,0))*12 from emp nvl(comm,0) 
--这个的意思是如果comm 为 null 那么就赋值为 0 和工资相加，如果不为null就使用comm的值相加。这个是因为在Oracle里面null默认是无穷大。

# INSTR(C1,C2,I,J)

I 搜索的开始位置,默认为1

J 出现的位置,默认为1

SQL> select instr(’oracle traning’,’ra’,1,2) instring from dual;

# 数字函数

# ABS(n)返回n的绝对值

​ SELECT ABS(100),ABS(-100) FROM dual;

# CEIL(n)向上取整

​ SELECT CEIL(3.1415927) FROM dual;

# EXP(n)返回数字e的n次方根

​ SELECT EXP(2),EXP(1) FROM dual;

# FLOOR(n)向下取整

​ SELECT FLOOR(30),FLOOR(15.63) FROM dual;

# LN(n)返回n的对数值

​ SELECT LN(1),LN(2),LN(2.7182818) FROM dual;

# LOG(n1, n)返回一个以n1为底n2的对数

​ SELECT LOG(2,1),LOG(2,4) FROM dual;

# MOD(n1, n2)返回一个n1除以n2的余数

​ SELECT MOD(10,3),MOD(3,3),MOD(2,3) FROM dual;

# POWER(n1, n2)返回n1的n2次方根

​ SELECT POWER(2,10),POWER(3,3) FROM dual;

# ROUND(n, int)对n按照int四舍五入

​ select round(55.55,1),round(-55.4),trunc(55.5),trunc(-55.3) from dual; -- 55.6 -55 55 -55

# TRUNC(n1, n2)按照n2对n1进行截取

​ select trunc(124.1666,-2) trunc1,trunc(124.16666,2) from dual;

# 转换函数

# TO_CHAR(x[,fmt]) 把日期和数字转换为指定格式的字符串

​ SELECT TO_CHAR(SYSDATE,'YYYY"年"MM"月"DD"日" HH24:MI:SS') FROM dual; SELECT TO_CHAR(-123456.45,'L9.9EEEEPR') FROM dual;

# TO_DATE(x, fmt)把字符串以fmt格式转换成一个日期格式

​ SELECT TO_DATE('2020-08-27 14:12:00', 'YYYY-MM-DD HH24:MI:SS') FROM dual;

# TO_NUMBER(x, fmt) 把字符串以fmt格式转换为一个数字

​ SELECT TO_NUMBER('-$12,345.67', '$99,999.99') FROM dual;

# NVL(expr1, expr2) 如果expr1为空，返回expr2，否则返回expr1

​ SELECT NVL(1, 100) FROM dual;

# NVL2(expr1, expr2, expr3) 如果expr1非空，返回expr2,否则返回expr3

​ SELECT NVL2(1, 100, 200) FROM dual;

# 日期函数

# ADD_MONTHS(date, int)增加或减去月份

​ select add_months(SYSDATE,2)，add_months(SYSDATE,-2) from dual;

# LAST_DAY(date)返回日期的最后一天

​ SELECT LAST_DAY(SYSDATE) FROM dual;

# ROUND(date, fmt),默认fmt为DDD，即月中的某一天。

​ SELECT SYSDATE 当时日期, ROUND(SYSDATE) 最近0点日期, ROUND(SYSDATE,'DAY') 最近星期日, ROUND(SYSDATE,'MONTH') 最近月初, ROUND(SYSDATE,'Q') 最近季初日期, ROUND(SYSDATE,'YEAR') 最近年初日期 FROM DUAL;

​ SELECT MONTHS_BETWEEN(SYSDATE, ADD_MONTHS(SYSDATE,2)) FROM dual;

# NEXT_DAY(date, ch)返回下周第一天

​ SELECT NEXT_DAY(SYSDATE,1) FROM dual;

/*

EXTRACT(fmt FROM d) 提取日期中特定的部分

fmt 为：YEAR、MONTH、DAY、HOUR、MINUTE、SECOND

YEAR、MONTH、DAY可以为 DATE 类型匹配，也可以与 TIMESTAMP 类型匹配；

但是 HOUR、MINUTE、SECOND 必须与 TIMESTAMP 类型匹配

*/

​ SELECT SYSDATE "date", EXTRACT(year from SYSDATE) year, EXTRACT(month from SYSDATE) month, EXTRACT(day from SYSDATE) day, EXTRACT(hour from SYSTIMESTAMP) hour, EXTRACT(minute from SYSTIMESTAMP) minute, EXTRACT(second from SYSTIMESTAMP) second FROM dual;

# Oracle中any和all的区别用法 (opens new window)

简单的说

All等价于N个And语句，Any等价于N个or语句。

对于any,all的用法,书中说的比较绕口,难以理解,如果通过举例就会比较清晰.

any的例子:

select * from t_hq_ryxx where gongz > any (select pingjgz from t_hq_bm);

输出的结果是所有大于字段'pingjgz'中最小数的值,简单来说就是输出的数全部大于'pingjgz'字段中的最小值;

select * from t_hq_ryxx where gongz < all (select pingjgz from t_hq_bm);

输出的结果是所有小于字段'pingjgz'中最大数的值,简单来说就是输出的数全部小于'pingjgz'字段中的最大值;

即:大于最小值,小于最大值

all的例子:

select * from t_hq_ryxx where gongz > all (select pingjgz from t_hq_bm);

输出的结果是所有大于字段'pingjgz'中最大数的值,简单来说就是输出的数全部大于'pingjgz'字段中的最大值;

select * from t_hq_ryxx where gongz < all (select pingjgz from t_hq_bm);

输出的结果是所有小于字段'pingjgz'中最小数的值,简单来说就是输出的数全部小于'pingjgz'字段中的最小值;

即:小于最小值,大于最大值

如果还是不清楚,把下面这句看懂也就够了:

any 就是匹配集合中的任意一个就满足条件了；而 all 要跟所有的都比较，所有都满足以后才为真。
